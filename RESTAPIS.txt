POST Login
IN: string username, string password
OUT: 200= "Login Success"- "Access Token" -> { string access_token,
				      	int expires_in; }
ERR: 401= "Incorrect Credentials"


POST ForgotPassword
IN: string username
OUT: 202= "Accepted"
ERR: 404= "User not found"

POST ChangePassword
IN string token, string oldPassword, string newPassword
OUT: 204 = “No Content”

POST RegisterUser
IN: string username, string password, string email
OUT: 200= "User registered"
ERR: 412= "Required property is missing"

POST CheckToken
IN: string token
OUT: string username
ERR: 401 = "Invalid token"
